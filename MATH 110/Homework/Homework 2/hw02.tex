	\section{Linear Independence}
	\begin{hw}
		Suppose $v_{1}, \ldots, v_{m}$ are linearly independent in $V$ and $w \in V$. Prove that 
		\begin{equation*}
			\dim{\Span (v_{1} - w, v_{2} - w, \ldots, v_{m} - w)} \geq m-1
		\end{equation*}
	\end{hw}

	\begin{solution}
		We first observe that since $v_{1}, \ldots, v_{m}$ are linearly independent, then that means that $\dim{\Span (v_{1}, v_{2}, \ldots, v_{m})} = m$. Furthermore, since $\Span (w)$ is the span of only one vector, it follows then that $\dim{\Span (w)} \leq 1$.
